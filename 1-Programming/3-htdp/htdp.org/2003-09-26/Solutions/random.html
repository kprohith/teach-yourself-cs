;; random-n-m : integer integer -> integer
;; to compute a random number between n and m-1 (inclusive)
;; Assume: n < m 
(define (random-n-m n m) 
  (+ (random (- m n)) n)) 

;; (random n) returns a number in this interval:
;; ------[--------------)--------
;;   -1  0  1  2  ...   n

;;   -1  0  1  2  ...  m-n

;; ------[-------------------)----
