;; repeat : natural-number -> list-of-symbols
(define (repeat n a-symbol)
  (cond
    [(zero? n) empty]
    [else (cons a-symbol (repeat (sub1 n) a-symbol))]))

;; Examples
(repeat 0 'doll) "should be" empty
(repeat 2 'rocket) "should be" (cons 'rocket (cons 'rocket empty))
