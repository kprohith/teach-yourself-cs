;; line-from-two-points : posn posn -> (number -> number)
;; to compute the line passing thru the two points p1 and p2
(define (line-from-two-points p1 p2)
  (cond
    [(= (posn-y p1) (posn-y p2))
     (error 'line-from-points "cannot build vertical lines")]
    [else
     (local [(define m (/ (- (posn-y p1) (posn-y p2))
                          (- (posn-x p1) (posn-x p2))))
             (define b (- (posn-y p1) (* m (posn-x p1))))
             (define (y x)
               (+ (* m x) b))]
       y)]))

;; tests
(graph-fun (line-from-two-points (make-posn 0 0) (make-posn 1 1)) 'red)
