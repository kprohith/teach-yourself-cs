;; determines if two functions are equal on the inputs 1.2, 3, and -5.7
(define (a-function=? f1 f2)
  (and (= (f1 1.2) (f2 1.2))
       (= (f1 2) (f2 2))
       (= (f1 -5.7) (f2 -5.7))))
