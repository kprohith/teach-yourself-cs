;; create-prices : number -> list-of-numbers
;; to create a list of prices, between $0.10 and $10.00
;; with increments of $0.10
(define (create-prices n)
  (cond
    [(zero? n) empty]
    [else (cons (* (+ (random 100) 1) 0.10)
                (create-prices (sub1 n)))]))

;; EXAMPLES
(create-prices 3)
;; = a list of three prices between $0.10 and $10.00
