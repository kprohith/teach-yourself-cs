;; line-form-point+slope : posn number -> (number -> number)
;; to construct a line function from a point on the line and its slope
(define (line-from-point+slope p m)
  (local [(define b (- (posn-y p) (* m (posn-x p))))
          (define (y x)
            (+ (* m x) b))]
    y))

;; tests
(graph-fun (line-from-point+slope (make-posn 0 4) 1) 'green)
(graph-fun (line-from-point+slope (make-posn 10 0) -1) 'blue)
(graph-fun (line-from-point+slope (make-posn 4 0) 1) 'green)
