;; tax : number -> number
;; to deteremine the amount of tax owed, for a given gross-pay
(define (tax gross-pay)
  (cond
    [(<= gross-pay 240) 0]
    [(<= gross-pay 480) (* gross-pay .15)]
    [else (* gross-pay .28)]))

;; EXAMPLES AS TESTS
(tax 10) "should be" 0
(tax 240) "should be" 0
(tax 300) "should be" 45
(tax 480) "should be" 72
(tax 500) "should be" 140

;; netpay : number -> number
;; to determine the amount of income, after taxes.
(define (netpay hours-worked)
  (- (* hours-worked 12) 
     (tax (* hours-worked 12))))

;; EXAMPLES AS TESTS
(netpay 1) "should be" 12
(netpay 40) "should be" 408
