(define (is-not-divisible-by<=i i n)
  (cond
    [(= 1 i) true]
    [else (and (not (= 0 (modulo n i)))
               (is-not-divisible-by<=i (sub1 i) n))]))

;; EXAMPLES AS TESTS
(is-not-divisible-by<=i 5 35) "should be" false
(is-not-divisible-by<=i 4 35) "should be" true

;; prime? : N -> boolean
;; to determine if n is prime
(define (prime? n)
  (cond
    [(= n 1) false]
    [else
     (is-not-divisible-by<=i (- n 1) n)]))

(prime? 1) "should be" false
(prime? 2) "should be" true
(prime? 3) "should be" true
(prime? 4) "should be" false
